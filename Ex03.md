%token ID

          ID '=' expression    /* Assignment: ID = expression */

        ;

          expression '+' ID
        | expression '-' ID
        | expression '*' ID
        | expression '/' ID
        | ID
        ;

%%

    return yyparse();
}

    fprintf(stderr, "Error: %s\n", s);
